package main

import (
	"fmt"
	"net"
)

func main() {
	conn, err := net.Dial("tcp", "localhost:8080") // Application Layer â†’ Transport Layer TCP
	if err != nil {
		panic(err)
	}
	defer conn.Close()

	message := "hello server" // Application Layer data
	_, err = conn.Write([]byte(message)) // Sends data to server
	if err != nil {
		panic(err)
	}

	fmt.Println("Message sent to server!")
}
